static inline uint16_t libusb_cpu_to_le16(const uint16_t x)
{
 union {
  uint8_t b8[2];
  uint16_t b16;
 } _tmp;
 _tmp.b8[1] = (uint8_t) (x >> 8);
 _tmp.b8[0] = (uint8_t) (x & 0xff);
 return _tmp.b16;
}

{

void __attribute__((visibility("default"))) sync_transfer_wait_for_completion(struct libusb_transfer *transfer)
{
 int r, *completed = transfer->user_data;
 struct libusb_context *ctx = ((((transfer->dev_handle)->dev)->ctx));

 while (!*completed) {
  r = libusb_handle_events_completed(ctx, completed);
  if (r < 0) {
   if (r == LIBUSB_ERROR_INTERRUPTED)
    continue;
   usbi_log(ctx, LIBUSB_LOG_LEVEL_ERROR, __FUNCTION__, "libusb_handle_events failed: %s, cancelling transfer and retrying", libusb_error_name(r))
                          ;
   libusb_cancel_transfer(transfer);
   continue;
  }
 }
}
